while (i < nrow(data1)) {
    activity_date <- as.character(data1[1, 1])
    mean_steps <- as.numeric(as.character(data1[1, 2]))
    j = 1
    while (j < nrow(data2) + 1) {
        d <- as.character(data2[j, 2])

            if (is.na(data2[j, 1])) {

            }

    }
